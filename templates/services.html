Here you can see the revealing pattern in use for a service

angular.module('angularPlay.services')
	.factory('HelloWorldService', 
		[function(){
			
			var initialValue = "Hi there"
			resetValue();
			
			var service = {
				"sayHello": sayHello,
				"set": updateValue,
				"get": getValue,
				"reset": resetValue
			};

			return service;

			function sayHello() {
				alert(alertValue);
			};

			function updateValue(value) {
				alertValue = value;
			};

			function getValue() {
				return alertValue;
			};

			function resetValue() {
				alertValue = initialValue;
			}
	}]);
